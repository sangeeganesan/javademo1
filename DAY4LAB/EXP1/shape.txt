package com.main;


	public abstract class Shape {
		  private String name;

		  public Shape(String name) {
		    this.name = name;
		  }

		  public String getName() {
		    return this.name;
		  }

		  public void setName(String name) {
		    this.name = name;
		  }
		  public abstract float calculateArea();
	}
